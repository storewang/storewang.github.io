4.2.1 生成证书文件

制作kubernetes ca证书

  1. 创建工作目录
mkdir /opt/k8s/etcd/{bin,cfg,ssl} -p
mkdir /opt/k8s/kubernetes/{bin,cfg,ssl} -p

  1. etcd ca配置

    • 2.1 /opt/k8s/kubernetes/ssl/ca-config.json

      {
   "signing": {
       "default": {
       "expiry": "87600h"
       },
       "profiles": {
       "kubernetes": {
           "expiry": "87600h",
           "usages": [
               "signing",
               "key encipherment",
               "server auth",
               "client auth"
           ]
       }
       }
   }
}
  2. 2.2 /opt/k8s/kubernetes/ssl/ca-csr.json
    {
  "CN": "kubernetes",
  "key": {
      "algo": "rsa",
      "size": 2048
  },
  "names": [
      {
          "C": "CN",
          "L": "Beijing",
          "ST": "Beijing",
          "O": "k8s",
          "OU": "System"
      }
  ]
}
  3. 2.3 生成kubernetes ca证书和私钥 初始化ca
    cfssl gencert -initca ca-csr.json | cfssljson -bare ca -
  4. 生成apiserver server证书
    • 3.1 apiserver server证书配置 /opt/k8s/kubernetes/ssl/server-csr.json
      {
"CN": "kubernetes",
"hosts": [
 "10.254.0.1",
 "127.0.0.1",
 "192.168.255.173",
 "192.168.255.152",
 "192.168.255.166",
 "kubernetes",
 "kubernetes.default",
 "kubernetes.default.svc",
 "kubernetes.default.svc.cluster",
 "kubernetes.default.svc.cluster.local"
],
"key": {
   "algo": "rsa",
   "size": 2048
},
"names": [
   {
       "C": "CN",
       "L": "Beijing",
       "ST": "Beijing",
       "O": "k8s",
       "OU": "System"
   }
]
}

  5. 3.2 生成server证书：apiserver

    cfssl gencert -ca=ca.pem -ca-key=ca-key.pem -config=ca-config.json -profile=kubernetes server-csr.json | cfssljson -bare server

  6. 制作kube-proxy证书

    • 4.1 kube-proxy证书配置 /opt/k8s/kubernetes/ssl/kube-proxy-csr.json

      {
"CN": "system:kube-proxy",
"hosts": [],
"key": {
   "algo": "rsa",
   "size": 2048
},
"names": [
   {
   "C": "CN",
   "L": "Beijing",
   "ST": "Beijing",
   "O": "k8s",
   "OU": "System"
   }
]
}

    • 4.2 生成server证书：kube-proxy

      cfssl gencert -ca=ca.pem -ca-key=ca-key.pem -config=ca-config.json -profile=kubernetes kube-proxy-csr.json | cfssljson -bare kube-proxy
